#include<math.h>
int max(int,int);

void warshal(int p[10][10],int n)
{
	 int i,j,k;
	 for(k=1;k<=n;k++)
	 for(i=1;i<=n;i++)
	 for(j=1;j<=n;j++)
	 p[i][j]=max(p[i][j],p[i][k]&&p[k][j]);
}

int max(int a,int b)
{                                                       ;
	if(a>b)
		return(a);
	else
		return(b);
}
