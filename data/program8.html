void bubble (int a[10],int n)
{
	for (int i=0; i<n; i++)
	{
		for (int j=0; j<n-i-1; j++)
		{
			if ( a[j] > a[j+1] )
			{
				int temp;
				temp = a[j];
				a[j] = a[j+1];
				a[j+1] = temp;
			}
		}
	}
	printf (" \n The sorted array using bubble sort is : \n \t ");
	for (i=0; i<n; i++)
	{
		printf("%d ",a[i]);
	}
}

void insertion (int a[10],int n)
{
	for ( int i=1 ; i<n ; i++ )
	{
		for (int j = 0 ; j < i ; j++ )
		{
			if ( a[j] > a[i] )
			{
				int temp = a[j] ;
				a[j] = a[i] ;

				for ( int k=i ; k>j ; k-- )
					a[k] = a[k-1] ;

				a[k+1] = temp ;
			}
		}
	}
	printf (" \n The sorted array using insertion sort is : \n \t ");
	for (i=0; i<n; i++)
	{
		printf("%d ",a[i]);
	}
}

void selection (int a[10],int n)
{
	for ( int i=0 ; i<n ; i++ )
	{
		for (int  j = i + 1 ; j < n ; j++ )
		{
			if ( a[i] > a[j] )
			{
				int temp = a[i] ;
				a[i] = a[j] ;
				a[j] = temp ;
			}
		}
	}

	printf (" \n The sorted array using selection sort is : \n \t ");
	for (i=0; i<n; i++)
	{
		printf("%d ",a[i]);
	}
}
