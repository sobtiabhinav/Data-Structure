#include <stdio.h> 
#include <conio.h>

void push(int st[],int data,int &top); 
void disp(int st[],int &top); 

int pop(int st[],int &top); 
int flg=0; 
int top=-1,tos=-1; 
int st[50]; 

void push(int st[],int data,int &top) 
{ 
if(top==50-1) 
flg=0; 
else 
{ 
flg=1; 
top++; 
st[top]=data; 
} 
} 

int pop(int st[],int &top) 
{ 
int pe; 
if(top==-1) 
{ 
pe=0; 
flg=0; 
} 
else 
{ 
flg=1; 
pe=st[top]; 
top--; 
} 
return(pe); 
}
 
void disp(int st[],int &top) 
{ 
int i; 
if(top==-1) 
{ 
printf("\nStack is Empty"); 
} 
else 
{ 
for(i=top;i>=0;i--) 
printf("\t%d",st[i]); 
} 
}

int dt,opt;

do 
{

switch(opt)

push(st,dt,tos); 
if(flg==1) 
{ 
printf("\nAfter Inserting the Element, Stack is:\n\n"); 
disp(st,tos); 
if(tos==50-1) 
printf("\nStack is Now Full"); 
} 
else 
printf("\nStack Overflow Insertion Not Possible"); 
break; 
case 2: 
dt=pop(st,tos); 
if(flg==1) 
{ 
printf("\n\tData Deleted From the Stack is:%d\n",dt);

else

case 3:

default:printf("\nWrong Choice Enter 1-3 Only");

}
