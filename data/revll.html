void AddElement(ch,*LLt )
{
	struct node *temp;
	temp=(struct node*)malloc(sizeof(struct node));
	temp->no=ch; 
	if (LLt==NULL)
	{
		LLt=temp;
		temp->next=NULL; 
	}
	else
	{ 
		LLt->next=temp;
		temp->next=NULL; 
	}
}

node *reverse(node *first)
{
	node *temp = NULL; 
	if(first->next != NULL)
	{
		temp = reverse(first->next);
		temp->next = first;
		return first;
	}
	else
		return first;
}
