int POM(int P[10],int x)
{
	int n,i,j,k,q,l,m[10][10],s[10][10];
	n=x;
	for(i=1;i<=n;i++)
	{
	m[i][i]=0;
	}
	for(l=2;l<=n;l++)
	{
		for(i=1;i<=n-l+1;i++)
		{
			j=i+l-1;
			m[i][j]=32000;
			for(k=i;k<=j-1;k++)
			{
				q=m[i][k]+m[k+1][j]+(P[i-1]*P[k]*P[j]);
				if(q<m[i][j])
				{
					m[i][j]=q;
					s[i][j]=k;
				}
			}
		}
	}
	pop(s,1,x);
	return s[10][10];
}
