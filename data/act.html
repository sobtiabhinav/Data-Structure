#include<stdio.h>
#include<conio.h>

act_sel(int a[20],int b[20],int x)
{
	int m,c[10],j,k;
	c[0]=1;k=1;j=1;
	for(m=2;m<=x;m++)
	{
		if(a[m]>=b[j])
		{
			c[k]=m;
			j=m;
			k=k+1;
		}
	}
	printf("\nThe possible activities are: ");
	for(j=0;j<k;j++)
	{
	printf("c%d ",c[j]);
	}
	return 0;
}

void main()
{
	int a[20],b[20],i,x;
	clrscr();
	printf("\nEnter number of activities: ");
	scanf("%d",&x);
	printf("\nEnter the starting time of activities: ");
	for(i=1;i<=x;i++)
	scanf("%d",&a[i]);
	printf("Enter the finish time of activities: ");
	for(i=1;i<=x;i++)
	scanf("%d",&b[i]);
	act_sel(a,b,x);
	getch();
}
