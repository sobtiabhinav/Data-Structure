#include <stdio.h>
#include <conio.h>

void display(int a[20], int n)
{
	for(int i=0; i<n; i++)
		printf("a[i]\t");
}

void main()
{
	int arr[20],i=0, n, opt, del, max, min;
	printf("How many elements? ");
	scanf("%d",&n);
	printf("\nEnter " n " elements : ");
	for (i=0; i<n; i++)
		scanf("%d",&arr[i]);
	do
	{
		printf("\n\n **MAIN MENU**");
		printf("\n1. Read and write the elements of an array");
		printf("\n2. Passing elements of an array to a function");
		printf("\n3. Delete an element from an array");
		printf("\n4. Traverse the elements an array");
		printf("\n5. Highest and lowest elements in an array of an array");
		printf("\n6. Exit");
		printf("\nEnter the operation to perform :") ;
		scanf("%d",&opt);
		switch (opt)
		{
			case 1:
			printf("\nThe entered array is :\n");
			for (i=0; i<n; i++)
				printf(arr[i]"\t");
			break;

			case 2:
			display(arr,n);
			break;

			case 3:
			printf("\nEnter the element number to delete: ");
			scanf("%d",&del);
			for (i=del-1; i<=n; i++)
				arr[i]= arr[i+1];
		    n=n-1;
			printf("\nThe omitted array is :\n");
			for (i=0; i<n; i++)
				printf(arr[i]"\t");
			break;

			case 4:
			printf("\nThe entered array in reverse order is :\n");      
			for (i=n-1; i>=0; i--)
				printf(arr[i]"\t");
			break;

			case 5:
			max=arr[0];
			min=arr[0];
			for (i=0; i<n; i++)
			{
				if (arr[i]>max)
				max=arr[i];
				else if (arr[i]<min)
				min=arr[i];
			}
			printf("\nHighest = ",max)
			printf("\nLowest = ",min);
			break;

			case 6:
			printf("\n Exiting Program !!");
			break;

			default:
			printf(" ERROR!! INVALID INPUT ");
			break;
		}
		getch();
	} while (opt!=6);
}
