#include <iostream.h>
#include <conio.h>
void display(int a[20], int n)
{
	for(int i=0; i<n; i++)
		cout<<a[i]<<"\t";
}

void main()
{
	int arr[20],i=0, n, opt, del, max, min;
	cout<<"How many elements? ";
	cin>>n;
	cout<<"\nEnter " <<n<< " elements : ";
	for (i=0; i<n; i++)
		cin>>arr[i];
	do
	{
	cout<<"\n\n **MAIN MENU**"
	    <<"\n1. Read and write the elements of an array"
	    <<"\n2. Passing elements of an array to a function"
	    <<"\n3. Delete an element from an array"
	    <<"\n4. Traverse the elements an array"
	    <<"\n5. Highest and lowest elements in an array of an array"
	    <<"\n6. Exit"
	    <<"\nEnter the operation to perform :" ;
	cin>>opt;
	switch (opt)
	{
		case 1:
		cout<<"\nThe entered array is :\n";
		for (i=0; i<n; i++)
		cout<<arr[i]<<"\t";
		break;

		case 2:
		display(arr,n);
		break;

		case 3:
		cout<<"\nEnter the element number to delete: ";
		cin>>del;
		for (i=del-1; i<=n; i++)
		arr[i]= arr[i+1];
		n=n-1;
		cout<<"\nThe omitted array is :\n";
		for (i=0; i<n; i++)
		cout<<arr[i]<<"\t";
		break;

		case 4:
		cout<<"\nThe entered array in reverse order is :\n";      
		for (i=n-1; i>=0; i--)
		cout<<arr[i]<<"\t";
		break;

		case 5:
		max=arr[0];
		min=arr[0];
		for (i=0; i<n; i++)
		{
			if (arr[i]>max)
			max=arr[i];
			else if (arr[i]<min)
			min=arr[i];
		}
		cout<<"\nHighest = "<<max
		    <<"\nLowest = "<<min;
		break;

		case 6:
		cout<<"\n Exiting Program !!";
		break;

		default:
		cout<<" ERROR!! INVALID INPUT ";
		break;
	}
	getch();
	} while (opt!=6);
}
