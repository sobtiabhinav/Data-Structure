#include<stdio.h>
#include<conio.h>
#include<malloc.h>

struct node
{
int info;
struct node*next;
};

typedef struct node NODE;
NODE*start;

void createemptylist(NODE*start)
{
start=NULL;
}

void displaylist(NODE*P)
{
printf("\nList is: ");
while(P!=NULL)
 {
printf("%d -> ",P->info);
P=P->next;
 }
printf("X\n");
}

void insertatbegin(int item)
{
NODE*P;
P=(NODE*)malloc(sizeof(NODE));
 if(P==NULL)
 {
printf("\nOVERFLOW!!!");
return;
 }
 else
 {
P->info=item;
  if(start==NULL)
P->next=NULL;
  else
P->next=start;
 }
start=P;
}

void insertatend(int item)
{
NODE*P,*loc;
P=(NODE*)malloc(sizeof(NODE));
 if(P==NULL)
 {
printf("\nOVERFLOW!!!");
return;
 }
P->info=item;
P->next=NULL;
 if(start==NULL)
start=P;
 else
 {
loc=start;
while(loc->next!=NULL)
loc=loc->next;
loc->next=P;
 }
}

void deletebeg()
{
NODE*P;
if(start==NULL)
{
printf("\nUNDERFLOW!!");
return;
}
P=start;
start=start->next;
printf("\nNumber deleted is %d \n",P->info);
free(P);
}

void deleteend()
{
NODE*P,*loc;
 if(start==NULL)
 {
printf("\nUNDERFLOW!!");
return;
 }
if(start->next==NULL)
{
P=start;
start=NULL;
}
else
{
loc=start;
P=start->next;
  while(P->next!=NULL)
  {
loc=P;
P=P->next;
  }
loc->next=NULL;
 }
printf("\nNumber deleted is %d \n",P->info);
free(P);
}

void main()
{
int choice,item,after;
char ch;
clrscr();
createemptylist(start);
printf("MENU:");
printf("\n1. Insert element at beginning.");
printf("\n2. Insert element at end.");
printf("\n3. Delete element from beginning.");
printf("\n4. Delete element from end.");
printf("\n5. Display the list.");
printf("\n6. Exit");
do
{
printf("\nEnter your choice: ");
scanf("%d",&choice);
  switch(choice)
  {
case 1: 
printf("\nInsert element at beginning.");
printf("\nEnter the item: ");
   scanf("%d",&item);
   insertatbegin(item);
   displaylist(start);
   break;
   case 2: 
   printf("\nInsert element at end.");
   printf("\nEnter the item: ");
   scanf("%d",&item);
   insertatend(item);
   displaylist(start);
   break;
   case 3: 
printf("\nDelete element from beginning.");
deletebeg();
displaylist(start);
break;
   case 4: 
printf("\nDelete element from end.");
deleteend();
   displaylist(start);
   break;
   case 5: 
printf("\nDisplay the list.");
   displaylist(start);
   break;
   case 6: 
return;
default: printf("\nINVALID CHOICE!!");
  }
fflush(stdin); //to clear memory
printf("\nDo you want to continue? (y/n): ");
scanf("%c",&ch);
 } while((ch=='y')||(ch=='Y'));

}
