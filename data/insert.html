#include <stdio.h>
#include <conio.h>
#include <time.h>

void main()
{
	int i,j,s,temp,a[20];
	clock_t top,bottom;
	clrscr();
	printf("Enter total elements: ");
	scanf("%d",&s);
	printf("Enter %d elements: ",s);
	for(i=0;i<s;i++)
		scanf("%d",&a[i]);
	top = clock();
	for(i=1;i<s;i++)
	{
		temp=a[i];
		j=i-1;
		while((temp<a[j])&&(j>=0))
		{
			a[j+1]=a[j];
			j=j-1;
		}
    a[j+1]=temp;
	}
	delay(100);
	bottom = clock();
	printf("After sorting: ");
	for(i=0;i<s;i++)
		printf(" %d",a[i]);
	printf("\n\nThe time complexity is : %f", (bottom-top)/CLK_TCK);
	getch();
}
