void mergesort(int x[],int n)
{
	int sub[MAX];

		{

	{
		list2=list1+size;
		u1=list2-1;
		u2=((list2+size-1)< n)?(list2+size-1):(n-1); 
		for(i=list1,j=list2;i< =u1 &&amp;amp;amp; j< =u2;k++)
		if(x[i]< =x[j])
		sub[k]=x[i++];
		else
		sub[k]=x[j++];
		for(;i< =u1;k++)
		sub[k]=x[i++];
		for(;j< =u2;k++)
		sub[k]=x[j++];
		list1=u2+1;
	}

		sub[k++] = x[i];
