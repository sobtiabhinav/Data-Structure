#include<stdio.h>

struct stack
{
	int info;
	struct stack *next;
};
typedef struct stack node;

class stlink
{
	node *start;
	public:
	stlink()
	{
		start=NULL;
	}
	void display(void);
	void push(int);
	int pop(void);
};

void stlink::push(int term)
{
	node *p,*s;
	s=start;
	if(s==NULL||s!=NULL)
	{
	p=(node *)malloc(sizeof(node));
	p->info=term;
	p->next=s;
	s=p;
	}
	start=s;
	return;
}

int stlink::pop(void)
{
	int term;
	if(start==NULL)
	{
		cout<<"UNDERFLOW";
		return(-1);
	}
	else
	{
		node *p;
		term=start->info;
		p=start;
		free(start);
		start=p->next;
		return(term);
	}
}
